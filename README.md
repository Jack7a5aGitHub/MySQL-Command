# MySQL-Command

#1 Setup local host
- mysql.server start
- mysql.server restart

# Connect to MySQL
- mysql -u <username> (Without Database Password)
- mysql -u <username> -p (With Database Password)

# Database
- show databases;
- use <database name>;

# Tables
- show tables;
- show create table <table name>;

# Procedure (Delimiters)
- show create procedure <procedure name>
